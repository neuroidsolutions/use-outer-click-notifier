```bash
npm install --save use-outer-click-notifier
```

```jsx
import React, { Component } from 'react'

import { useMyHook } from 'use-outer-click-notifier'

const Example = () => {
  const example = useMyHook()
  return (
    <div>{example}</div>
  )
}
```
